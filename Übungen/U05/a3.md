## (a)
Um die Balancierung des Baumes zu gewährleisten, wenn es viele Elemente mit gleichem Schlüssel gibt.


## (b)
findAlle(k, node=root, out=[]) returns node[]:
    if node == nil:
        return out
    else:
        if k == node.key:
            out += node
            if k == node.right.key:
                return findAlle(k, node.right, out)
            if k == node.left.key:
                return findAlle(k, node.left, out)
        elif k < node.key:
            return findAlle(k, node.left, out)
        elif k > node.key:
            return findAlle(k, node.right, out)
